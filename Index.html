import tkinter as tk from tkinter import messagebox import time class SimuladorVisualDCU: def __init__(self, root): self.root = root self.root.title("Simulador Visual PHVA - DCU (UNEMI)") self.root.geometry("700x550") self.root.configure(bg="#1e1e2e") self.satisfactorio = False self.iteracion = 1 titulo = tk.Label( root, text="Simulador Dinámico DCU-PHVA", font=("Arial", 16, "bold"), fg="#cdd6f4", bg="#1e1e2e" ) titulo.pack(pady=10) self.canvas = tk.Canvas( root, width=600, height=350, bg="#252538", highlightthickness=0 ) self.canvas.pack(pady=10) self.puntos = { "Planear": (100, 100), "Hacer": (500, 100), "Verificar": (500, 280), "Actuar": (100, 280) } self.dibujar_ciclo() x_inicio, y_inicio = self.puntos["Planear"] self.proceso_movil = self.canvas.create_oval( x_inicio - 12, y_inicio - 12, x_inicio + 12, y_inicio + 12, fill="#f5e0dc", outline="#f38ba8", width=3 ) self.lbl_estado = tk.Label( root, text="Estado: Listo para iniciar", font=("Arial", 12), fg="#a6adc8", bg="#1e1e2e" ) self.lbl_estado.pack(pady=5) self.frame_botones = tk.Frame(root, bg="#1e1e2e") self.frame_botones.pack(pady=15) self.btn_iniciar = tk.Button( self.frame_botones, text="Iniciar Iteración", font=("Arial", 11, "bold"), bg="#89b4fa", fg="#11111b", command=self.iniciar_ciclo ) self.btn_iniciar.pack() def dibujar_ciclo(self): self.canvas.create_line(100, 100, 500, 100, fill="#45475a", width=4, arrow=tk.LAST) self.canvas.create_line(500, 100, 500, 280, fill="#45475a", width=4, arrow=tk.LAST) self.canvas.create_line(500, 280, 100, 280, fill="#45475a", width=4, arrow=tk.LAST) self.canvas.create_line(100, 280, 100, 100, fill="#ba1818", width=4, arrow=tk.LAST, dash=(4, 4)) self.crear_nodo("Planear", 100, 100, "#a6e3a1") self.crear_nodo("Hacer", 500, 100, "#fab387") self.crear_nodo("Verificar", 500, 280, "#f9e2af") self.crear_nodo("Actuar", 100, 280, "#f38ba8") def crear_nodo(self, texto, x, y, color): self.canvas.create_rectangle( x - 50, y - 25, x + 50, y + 25, fill="#313244", outline=color, width=2 ) self.canvas.create_text( x, y, text=texto, fill="#cdd6f4", font=("Arial", 10, "bold") ) def mover_objeto(self, origen, destino, velocidad=15): x1, y1 = self.puntos[origen] x2, y2 = self.puntos[destino] pasos = 30 dx = (x2 - x1) / pasos dy = (y2 - y1) / pasos for _ in range(pasos): self.canvas.move(self.proceso_movil, dx, dy) self.root.update() time.sleep(1 / velocidad) def iniciar_ciclo(self): self.btn_iniciar.config(state=tk.DISABLED) x_p, y_p = self.puntos["Planear"] coords = self.canvas.coords(self.proceso_movil) self.canvas.move( self.proceso_movil, x_p - (coords[0] + 12), y_p - (coords[1] + 12) ) self.lbl_estado.config( text=f"Iteración {self.iteracion} - Planear: Analizando requisitos...", fg="#a6e3a1" ) self.root.update() time.sleep(1) self.mover_objeto("Planear", "Hacer") self.lbl_estado.config( text=f"Iteración {self.iteracion} - Hacer: Creando soluciones de diseño...", fg="#fab387" ) self.root.update() time.sleep(1) self.mover_objeto("Hacer", "Verificar") self.lbl_estado.config( text=f"Iteración {self.iteracion} - Verificar: ¿Satisface el diseño al usuario?", fg="#f9e2af" ) self.root.update() respuesta = messagebox.askyesno( "Evaluación EDR", "¿Satisface el sistema los requerimientos del usuario?" ) self.mover_objeto("Verificar", "Actuar") if respuesta: self.lbl_estado.config( text="Actuar: ¡Desarrollo Exitoso! Proceso terminado.", fg="#a6e3a1" ) messagebox.showinfo( "Éxito", f"¡Sistema aprobado con éxito en la iteración {self.iteracion}!" ) self.satisfactorio = True self.root.destroy() else: self.lbl_estado.config( text="Actuar: Requisitos NO satisfechos. Regresando al inicio...", fg="#f38ba8" ) messagebox.showwarning( "Fallo en Evaluación", "Los requerimientos no se cumplieron. El ciclo vuelve a Planear." ) self.mover_objeto("Actuar", "Planear", velocidad=40) self.iteracion += 1 self.root.after(1000, self.iniciar_ciclo) if __name__ == "__main__": ventana = tk.Tk() app = SimuladorVisualDCU(ventana) ventana.mainloop()
